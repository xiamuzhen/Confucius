cmake_minimum_required(VERSION 3.25)
project(untitled C)

set(CMAKE_C_STANDARD 11)

file(GLOB BASE Base/*.c)
file(GLOB LEETCODE Leetcode/*.c)

foreach (PATH ${BASE})
    STRING(REGEX REPLACE ".+/(.+)\\..*" "\\1" NAME ${PATH})
    add_executable(${NAME} ${PATH})
endforeach ()

foreach (PATH ${LEETCODE})
    STRING(REGEX REPLACE ".+/(.+)\\..*" "\\1" NAME ${PATH})
    add_executable(${NAME} ${PATH})
endforeach ()
