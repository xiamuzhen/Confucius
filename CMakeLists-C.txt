cmake_minimum_required(VERSION 3.25)
project(untitled C)

set(CMAKE_C_STANDARD 11)

file(GLOB C_BASE Base/*.c)
file(GLOB C_LEETCODE Leetcode/*.c)

foreach(FILE_PATH ${C_BASE})
    STRING(REGEX REPLACE ".+/(.+)\\..*" "\\1" FILE_NAME ${FILE_PATH})
    add_executable(${FILE_NAME} ${FILE_PATH} )
endforeach()

foreach(FILE_PATH ${C_LEETCODE})
    STRING(REGEX REPLACE ".+/(.+)\\..*" "\\1" FILE_NAME ${FILE_PATH})
    add_executable(${FILE_NAME} ${FILE_PATH} )
endforeach()
